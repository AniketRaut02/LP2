# N-Queen Problem using Backtracking

# Function to print board
def print_board(board, n):

    print("\nSolution:\n")

    for i in range(n):
        for j in range(n):

            if board[i][j] == 1:
                print("Q", end=" ")

            else:
                print(".", end=" ")

        print()


# Function to check safe position
def is_safe(board, row, col, n):

    # Check left side row
    for i in range(col):
        if board[row][i] == 1:
            return False

    # Check upper diagonal
    i = row
    j = col

    while i >= 0 and j >= 0:

        if board[i][j] == 1:
            return False

        i -= 1
        j -= 1

    # Check lower diagonal
    i = row
    j = col

    while i < n and j >= 0:

        if board[i][j] == 1:
            return False

        i += 1
        j -= 1

    return True


# Backtracking function
def solve_nqueen(board, col, n):

    # If all queens placed
    if col >= n:
        return True

    # Try every row
    for i in range(n):

        if is_safe(board, i, col, n):

            board[i][col] = 1

            # Recursive call
            if solve_nqueen(board, col + 1, n):
                return True

            # Backtracking
            board[i][col] = 0

    return False


# -------- MAIN PROGRAM --------

n = int(input("Enter number of queens: "))

# Create board
board = []

for i in range(n):
    row = [0] * n
    board.append(row)

# Solve problem
if solve_nqueen(board, 0, n):

    print_board(board, n)

else:
    print("Solution does not exist")
