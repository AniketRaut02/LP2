# Simplified DES Implementation

# Initial Permutation Table
IP = [2,6,3,1,4,8,5,7]

# Final Permutation Table
FP = [4,1,3,5,7,2,8,6]

# Expansion Table
EP = [4,1,2,3,2,3,4,1]

# P4 Table
P4 = [2,4,3,1]

# S-Boxes
S0 = [[1,0,3,2],
      [3,2,1,0],
      [0,2,1,3],
      [3,1,3,2]]

S1 = [[0,1,2,3],
      [2,0,1,3],
      [3,0,1,0],
      [2,1,0,3]]


# Permutation function
def permute(bits, table):
    return ''.join(bits[i-1] for i in table)


# XOR function
def xor(a, b):
    return ''.join('0' if a[i]==b[i] else '1' for i in range(len(a)))


# S-Box lookup
def sbox(bits, box):
    row = int(bits[0] + bits[3], 2)
    col = int(bits[1] + bits[2], 2)
    return format(box[row][col], '02b')


# DES round function
def fk(bits, key):
    left = bits[:4]
    right = bits[4:]

    temp = permute(right, EP)
    temp = xor(temp, key)

    left_s0 = sbox(temp[:4], S0)
    right_s1 = sbox(temp[4:], S1)

    temp = permute(left_s0 + right_s1, P4)

    left = xor(left, temp)

    return left + right


# Encryption
plain = input("Enter 8-bit Plain Text: ")
key = input("Enter 8-bit Key: ")

# Initial Permutation
bits = permute(plain, IP)

# Round 1
bits = fk(bits, key)

# Swap
bits = bits[4:] + bits[:4]

# Round 2
bits = fk(bits, key)

# Final Permutation
cipher = permute(bits, FP)

print("Encrypted Text:", cipher)
